import React from 'react'

## Install

```bash
npm install @connector-kit/connector
```

## Quick Start

Wrap your app with `ConnectorProvider` and add a connect button:

```tsx title="App.tsx"
"use client"

import { ConnectorProvider, ConnectButton, getDefaultConfig } from '@connector-kit/connector'

export default function App() {
  return (
    <ConnectorProvider config={getDefaultConfig({ appName: "Your App" })}>
      <ConnectButton />
    </ConnectorProvider>
  )
}
```

## Access Wallet State

Use the `useConnector` hook to access wallet connection state and methods:

```tsx title="WalletInfo.tsx"
import { useConnector } from '@connector-kit/connector'

function WalletInfo() {
  const { connected, selectedWallet, selectedAccount, disconnect } = useConnector()

  if (!connected) return <div>Not connected</div>
  
  return (
    <div>
      <p>Wallet: {selectedWallet?.name}</p>
      <p>Account: {selectedAccount}</p>
      <button onClick={disconnect}>Disconnect</button>
    </div>
  )
}
```

## Configuration

Configure the provider with `getDefaultConfig()` helper:

```tsx
import { getDefaultConfig, getDefaultMobileConfig } from '@connector-kit/connector'

const config = getDefaultConfig({
  appName: "Your App",
  appUrl: "https://your-app.com",
  autoConnect: true,      // Auto-reconnect to last wallet
  debug: false,           // Enable debug logging
  network: 'mainnet-beta' // Solana network
})

const mobile = getDefaultMobileConfig({
  appName: "Your App", 
  appUrl: "https://your-app.com",
  network: 'mainnet-beta'
})

<ConnectorProvider config={config} mobile={mobile}>
  {children}
</ConnectorProvider>
```

## Components

| Component | Description |
|-----------|-------------|
| `ConnectorProvider` | Root provider for wallet state management |
| `ConnectButton` | Complete wallet connection interface with dropdown |
| `ConnectModal` | Standalone wallet selection modal |

## Hooks

| Hook | Returns | Description |
|------|---------|-------------|
| `useConnector()` | `ConnectorSnapshot` | Main wallet connection state and methods |
| `useConnectorClient()` | `ConnectorClient` | Access to underlying client instance |
| `useModal()` | `UseModalReturn` | Modal state management and navigation |

### useConnector Properties

| Property | Type | Description |
|----------|------|-------------|
| `wallets` | `WalletInfo[]` | Available wallets |
| `selectedWallet` | `Wallet \| null` | Connected wallet |
| `connected` | `boolean` | Connection status |
| `connecting` | `boolean` | Connection in progress |
| `accounts` | `AccountInfo[]` | Available accounts |
| `selectedAccount` | `string \| null` | Selected account address |
| `select(name)` | `(name: string) => Promise<void>` | Connect to wallet |
| `disconnect()` | `() => Promise<void>` | Disconnect wallet |
| `selectAccount(address)` | `(address: string) => Promise<void>` | Switch account |
| `modalOpen` | `boolean` | Modal visibility state |
| `openModal()` | `() => void` | Open connection modal |
| `closeModal()` | `() => void` | Close connection modal |

## SDK Integration

Add the SDK for Solana operations:

```bash
npm install @connector-kit/sdk
```

Use together with connector for complete functionality:

```tsx title="BalanceDisplay.tsx"
import { useWalletAddress, useBalance } from '@connector-kit/sdk'

function BalanceDisplay() {
  const { address, connected } = useWalletAddress()
  const { balance } = useBalance({ address })

  if (!connected) return <div>Connect wallet to view balance</div>

  return (
    <div>
      <p>Address: {address}</p>
      <p>Balance: {balance ? Number(balance) / 1e9 : 0} SOL</p>
    </div>
  )
}
```
