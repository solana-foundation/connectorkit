## 3. Headless Implementation

Complete control over UI and behavior:

```tsx
import { useConnector } from '@connector-kit/connector'
import { useWalletAddress, useBalance } from '@connector-kit/sdk'

function CustomWalletConnect() {
  const { wallets, select, disconnect, connected, connecting } = useConnector()
  const { address } = useWalletAddress()
  const { balance } = useBalance({ address })

  return (
    <div>
      {!connected ? (
        <div>
          <h3>Available Wallets</h3>
          {wallets?.map(wallet => (
            <button 
              key={wallet.name}
              onClick={() => select(wallet.name)}
              disabled={!wallet.installed || connecting}
            >
              {wallet.installed ? (
                <>
                  {wallet.icon && <img src={wallet.icon} alt="" width={16} height={16} />}
                  Connect {wallet.name}
                </>
              ) : (
                `${wallet.name} (Not Installed)`
              )}
            </button>
          ))}
        </div>
      ) : (
        <div>
          <p>Connected: {address}</p>
          <p>Balance: {balance ? Number(balance) / 1e9 : 0} SOL</p>
          <button onClick={disconnect}>Disconnect</button>
        </div>
      )}
    </div>
  )
}
```

## 4. Advanced Features

Access wallet capabilities and modal navigation:

```tsx
import { useConnector, useModal } from '@connector-kit/connector'
import { useTransaction } from '@connector-kit/sdk'

function AdvancedWalletFeatures() {
  const { selectedWallet, connected } = useConnector()
  const modal = useModal()
  const { sendTransaction } = useTransaction()
  
  const handleOpenProfile = () => {
    modal.openProfile() // Opens profile modal
  }
  
  const handleOpenSettings = () => {
    modal.openSettings() // Opens settings modal
  }

  const handleSendSOL = async () => {
    try {
      // Create a simple transfer transaction
      const signature = await sendTransaction({
        to: 'recipient-address',
        amount: 0.001 * 1e9 // 0.001 SOL in lamports
      })
      console.log('Transaction sent:', signature)
    } catch (error) {
      console.error('Transaction failed:', error)
    }
  }
  
  if (!connected) return <div>Connect wallet to access features</div>
  
  return (
    <div>
      <h3>Wallet: {selectedWallet?.name}</h3>
      <button onClick={handleOpenProfile}>View Profile</button>
      <button onClick={handleOpenSettings}>Settings</button>
      <button onClick={handleSendSOL}>Send 0.001 SOL</button>
    </div>
  )
}
```

## Complete Setup

For full functionality, install both packages:

```bash
npm install @connector-kit/connector @connector-kit/sdk
```

Setup providers at your app root:

```tsx title="app/layout.tsx"
import { ConnectorProvider, getDefaultConfig } from '@connector-kit/connector'
import { ArcProvider } from '@connector-kit/sdk'

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html>
      <body>
        <ConnectorProvider config={getDefaultConfig({ appName: "Your App" })}>
          <ArcProvider network="mainnet-beta">
            {children}
          </ArcProvider>
        </ConnectorProvider>
      </body>
    </html>
  )
}
```
